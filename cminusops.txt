C-- Programming Language Specification

Program			−→ 	VarDeclList FunDeclList
VarDeclList		−→ 	[none]
				VarDecl VarDeclList
				VarDecl
VarDecl			−→ 	Type id ;
				Type id [ num ] ;
FunDeclList		−→	FunDecl
				FunDecl FunDeclList
FunDecl			−→	Type id ( ParamDeclList ) Block
ParamDeclList		−→	[none]
				ParamDeclListTail
ParamDeclListTail	−→	ParamDecl
				ParamDecl, ParamDeclListTail
ParamDecl		−→	Type id
				Type id []
Block			−→	{ VarDeclList StmtList }
Type			−→	int
				char
StmtList		−→	Stmt
				Stmt StmtList
Stmt			−→	;
				Expr ;
				return Expr ;
				read id ;
				write Expr ;
				writeln ;
				break ;
				if ( Expr ) Stmt else Stmt
				while ( Expr ) Stmt
				Block
Expr			−→	UnaryOp Expr
				Expr BinOp Expr
				id = Expr
				id [ Expr ] = Expr
				// Primary types, merged with Expr
				id
				num
				( Expr )
				id ( ExprList )
				id [ Expr ]
ExprList		−→	[none]
				ExprListTail
ExprListTail		−→	Expr
				Expr, ExprListTail
UnaryOp			−→	− | !
BinOp			−→ 	+|−|*|/|==|!=|<|<=|>|>=&&|||
